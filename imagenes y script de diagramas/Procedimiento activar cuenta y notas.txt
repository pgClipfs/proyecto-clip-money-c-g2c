CREATE PROCEDURE [dbo].[proc_activar_cuenta]
	@frenteDNI varchar(MAX),
	@reversoDNI varchar(MAX),
	@idUsuario int
AS
BEGIN
	SET NOCOUNT ON;
		BEGIN TRY
			BEGIN TRANSACTION
				
				UPDATE usuarios
				   SET 
						foto_frente_dni = @frenteDNI,
						foto_reverso_dni = @reversoDNI
				 WHERE id_usuario=@idUsuario
			
				UPDATE cuentas
				   SET 
					  cuenta_activa = 1
				 WHERE id_us=@idUsuario
			COMMIT TRANSACTION
		END TRY
		BEGIN CATCH
			RAISERROR('An error occurred', 16/*error severity*/, 1);
			ROLLBACK TRANSACTION
		END CATCH
END
